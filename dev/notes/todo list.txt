		if (isNaN(obj)) {
			return arr;
		}
		if( parts.length == 2 ){
				top = Ratio.parseNumber(parts[0]);
				j = Math.abs(+obj) < 1;
				arr[ j ? 0 : 1] = top[ j ? 0 : 1];
				arr[ j ? 1 : 0] =  +(top[ j ? 1 : 0 ] + "e" + Math.abs(+parts[1]));

		}
		return arr;
